# K8s + sonarqube yaml로 배포하기

## Namespace 생성
  * vi sonarqube-ns.yml

~~~~yaml
apiVersion: v1
kind: Namespace
metadata:
    name: sonarqube-ns
~~~~

## StorageClass 생성
  * vi sonarqube-sc.yml

~~~~yaml
apiVersion: storage.k8s.io/v1
kind: StorageClass
metadata:
  name: sonarqube-storage
provisioner: kubernetes.io/no-provisioner
#volumeBindingMode: WaitForFirstConsumer
~~~~

## Secret 생성
  * vi sonarqube-secret.yaml

~~~~yaml
apiVersion: v1
kind: Secret
metadata:
  name: postgres-pwd
  namespace: sonarqube-ns
type: kubernetes.io/basic-auth
stringData:
  password: P@s$w0rd # required field for kubernetes.io/basic-authstrato@k8s-master:~/~/SonarQube$ ^C
~~~~

## PV 생성
  * vi sonarqube-pv.yml

~~~~yaml
apiVersion: v1
kind: PersistentVolume
metadata:
  name: sonar-postgresql-pv
spec:
  storageClassName: sonarqube-storage
  accessModes:
  - ReadWriteOnce
  capacity:
    storage: 5Gi
  persistentVolumeReclaimPolicy: Retain
  local:                                         #hostpath는 단일노드테스트전용.  local 볼륨 사용 할것
    path: /data/sonarqube
  nodeAffinity:                                    #master 노드에 생성
    required:
      nodeSelectorTerms:
        - matchExpressions:
          - key: kubernetes.io/hostname                           #- key: kubernetes.io/hostname (노드명으로 할때)
            operator: In
            values:
            - k8s-node1

~~~~

## PVC 생성
  * vi sonarqube-pvc.yml

~~~~yaml
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: sonarqube-postgresql-pvc
  namespace: sonarqube-ns
spec:
  storageClassName: sonarqube-storage
  volumeName: sonar-postgresql-pv       # 연결할 PV 이름
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 5Gi
~~~~

## Deployment 생성 (postgres)
  * vi sonarqube-progres-deployment.yml

~~~~yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: sonarqube-postgres
  namespace: sonarqube-ns
spec:
  replicas: 1
  selector:
    matchLabels:
      app: sonarqube-postgres
  template:
    metadata:
      labels:
        app: sonarqube-postgres
    spec:
      containers:
        - image: postgres:14.5
          name: sonarqube-postgres
          env:
            - name: POSTGRES_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: postgres-pwd
                  key: password
            - name: POSTGRES_USER
              value: sonar
          ports:
            - containerPort: 5432
              name: postgresport
          volumeMounts:
            # This name must match the volumes.name below.
            - name: postgres-data
              mountPath: /var/lib/postgresql/data
              subPath: postgres-data
      volumes:
        - name: postgres-data
          persistentVolumeClaim:
            claimName: sonarqube-postgresql-pvc
~~~~

## Service 생성 (postgres)

apiVersion: v1

  namespace: sonarqube-ns

  selector:
    app: sonarqube-postgres
  ports:
    - port: 5432
      targetPort: 5432
      protocol: TCP
~~~~
